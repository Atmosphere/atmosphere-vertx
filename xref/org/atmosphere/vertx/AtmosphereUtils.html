35      public final static AtmosphereRequest request(final HttpServerRequest request) throws Throwable {
36          final String base = getBaseUri(request);
37          final URI requestUri = new URI(base.substring(0, base.length() - 1) + request.uri());
38          String ct = "text/plain";
39          if (request.headers().get("Content-Type") != null) {
40              ct = request.headers().get("Content-Type");
41          }
42          String method = request.method();
43  
44  
45          URI uri = null;
46          try {
47              uri = URI.create(request.uri());
48          } catch (IllegalArgumentException e) {
49              logger.trace("", e);
50          }
51          String queryString = uri.getQuery();
52          Map<String, String[]> qs = new HashMap<String, String[]>();
53          if (queryString != null) {
54              parseQueryString(qs, queryString);
55          }
56  
57          String u = requestUri.toURL().toString();
58          int last = u.indexOf("?") == -1 ? u.length() : u.indexOf("?");
59          String url = u.substring(0, last);
60          int l = requestUri.getAuthority().length() + requestUri.getScheme().length() + 3;
61  
62          final Map<String, Object> attributes = new HashMap<String, Object>();
63  
64          final StringBuilder b = new StringBuilder();
65  
66          int port = uri == null ? 0 : uri.getPort();
67          String uriString = uri.getPath();
68          String host = uri.getHost();
69          AtmosphereRequest.Builder requestBuilder = new AtmosphereRequest.Builder();
70          final AtmosphereRequest r = requestBuilder.requestURI(url.substring(l))
71                  .requestURL(u)
72                  .pathInfo(url.substring(l))
73                  .headers(getHeaders(request))
74                  .method(method)
75                  .requestURL(request.uri())
76                  .contentType(ct)
77                  .destroyable(false)
78                  .attributes(attributes)
79                  .servletPath("")
80                  .remotePort(port)
81                  .remoteAddr(uriString)
82                  .remoteHost(host)
83  //                .localPort(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getPort())
84  //                .localAddr(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getAddress().getHostAddress())
85  //                .localName(((InetSocketAddress) ctx.getChannel().getLocalAddress()).getHostName())
86                  .body(b.toString())
87                  .queryStrings(qs)
88                  .build();
89          return r;
90      }
91  
92      public static void parseQueryString(Map<String, String[]> qs, String queryString) {
93          if (queryString != null) {
94              String[] s = queryString.split("&");
95              for (String a : s) {
96                  String[] q = a.split("=");
97                  String[] z = new String[]{q.length > 1 ? q[1] : ""};
98                  qs.put(q[0], z);
99              }
100         }
101     }
102 
103     public static String getBaseUri(final HttpServerRequest request) {
104         return "http://" + request.headers().get(HttpHeaders.Names.HOST) + "/";
105 
106     }
107 
108     public static Map<String, String> getHeaders(final HttpServerRequest request) {
109         final Map<String, String> headers = new HashMap<String, String>();
110 
111         for (Entry<String, String> e : request.headers()) {
112             headers.put(e.getKey().toLowerCase(), e.getValue());
113         }
114 
115         return headers;
116     }
