37  /**
38   * A boostrap class that can be used to bridge Atmosphere and Vert.x. As simple as
39   * <pre><blockquote>
40   public class VertxJerseyChat extends Verticle {
41  
42       private static final Logger logger = LoggerFactory.getLogger(VertxJerseyChat.class);
43  
44       @Override
45       public void start() throws Exception {
46           VertxAtmosphere.Builder b = new VertxAtmosphere.Builder();
47           HttpServer httpServer = vertx.createHttpServer();
48  
49           httpServer.requestHandler(new Handler&gt;HttpServerRequest&lt;() {
50               public void handle(HttpServerRequest req) {
51                   String path = req.path;
52                   if (path.equals("/")) {
53                       path = "/index.html";
54                   }
55  
56                   logger.info("Servicing request {}", path);
57                   req.response.sendFile("src/main/resources" + path);
58               }
59           });
60  
61           b.resource(ResourceChat.class)
62            .initParam(ApplicationConfig.WEBSOCKET_CONTENT_TYPE, "application/json")
63            .httpServer(httpServer).url("/chat").build();
64  
65           httpServer.listen(8080);
66       }
67   }
68   * </blockquote></pre>
69   * @author  Jeanfrancois Arcand
70   */
71  public class VertxAtmosphere {
72      private static final Logger logger = LoggerFactory.getLogger(VertxAtmosphere.class);
73  
74      private final Builder b;
75      private final AtmosphereCoordinator coordinator = new AtmosphereCoordinator();
76  
77      private VertxAtmosphere(Builder b) {
78          this.b = b;
79  
80          RouteMatcher routeMatcher = new RouteMatcher();
81          routeMatcher.get(b.url, handleHttp());
82          routeMatcher.post(b.url, handleHttp());
83          routeMatcher.noMatch(b.httpServer.requestHandler());
84  
85          b.httpServer.requestHandler(routeMatcher);
86          b.httpServer.websocketHandler(handleWebSocket());
87  
88          if (b.resource != null) {
89              coordinator.configure(b);
90          }
91          coordinator.ready();
92      }
93

96       * @param path

98       */
99      public boolean matchPath(String path) {

222     private Handler<HttpServerRequest> handleHttp() {
223 
224         return new Handler<HttpServerRequest>() {
225             @Override
226             public void handle(HttpServerRequest req) {
227                 logger.trace("HTTP received");
228                 coordinator.route(req);
229             }
230         };
231     }
232 
233     private Handler<ServerWebSocket> handleWebSocket() {
234         return new Handler<ServerWebSocket>() {
235             @Override
236             public void handle(ServerWebSocket webSocket) {
237                 logger.trace("WebSocket received {}", webSocket);
238                 coordinator.route(webSocket);
239             }
240         };
