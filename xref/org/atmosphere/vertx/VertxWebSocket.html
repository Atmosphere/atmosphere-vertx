16  package org.atmosphere.vertx;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  import org.vertx.java.core.http.ServerWebSocket;
22  
23  import java.io.IOException;
24  import java.util.concurrent.atomic.AtomicBoolean;
25  
26  /**
27   * An Atmosphere's {@link org.atmosphere.websocket.WebSocket} wrapper around a {@link ServerWebSocket}
28   *
29   * @author Jeanfrancois Arcand
30   */
31  public class VertxWebSocket extends org.atmosphere.websocket.WebSocket {
32      private static final Logger logger = LoggerFactory.getLogger(VertxWebSocket.class);
33      private final AtomicBoolean isOpen = new AtomicBoolean(true);
34      private final ServerWebSocket webSocket;
35  
36      public VertxWebSocket(final AtmosphereConfig config, final ServerWebSocket webSocket) {
37          super(config);
38          this.webSocket = webSocket;
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      public org.atmosphere.websocket.WebSocket write(String data) throws IOException {
46          logger.trace("WebSocket.write()");
47          webSocket.writeTextFrame(data);
48          lastWrite = System.currentTimeMillis();
49          return this;
50      }
51  
52      /**
53       * {@inheritDoc}
54       */
55      @Override
56      public org.atmosphere.websocket.WebSocket write(byte[] data, int offset, int length) throws IOException {
57          webSocket.writeTextFrame(new String(data, offset, length, "UTF-8"));
58          return this;
59      }
60  
61      /**
62       * {@inheritDoc}
63       */
64      @Override
65      public boolean isOpen() {
66          return isOpen.get();
67      }
68  
69      /**
70       * {@inheritDoc}
71       */
72      @Override
73      public void close() {
74          if (!isOpen()) {
75              isOpen.set(false);
76              webSocket.close();
77          }
78      }
79  }
