24  import org.atmosphere.cpr.AtmosphereResourceFactory;

26  import org.atmosphere.cpr.MetaBroadcaster;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;

30  import java.io.IOException;
31  import java.util.concurrent.ConcurrentHashMap;

33  /**
34   * Simple annotated class that demonstrate the power of Atmosphere. This class supports all transports, support
35   * message length garantee, heart beat, message cache thanks to the @managedAService.
36   */
37  @ManagedService(path = "chat/{room: [a-zA-Z][a-zA-Z_0-9]*}")
38  public class ChatRoom {
39      private final Logger logger = LoggerFactory.getLogger(ChatRoom.class);
40  
41      private final ConcurrentHashMap<String, String> users = new ConcurrentHashMap<String, String>();
42      private String chatroomName;
43      private String mappedPath;
44      private BroadcasterFactory factory;
45  
46      /**
47       * Invoked when the connection as been fully established and suspended, e.g ready for receiving messages.
48       *
49       * @param r
50       */
51      @Ready(value = Ready.DELIVER_TO.ALL, encoders = {JacksonEncoder.class})
52      public ChatProtocol onReady(final AtmosphereResource r) {
53          logger.info("Browser {} connected.", r.uuid());
54          if (chatroomName == null) {
55              mappedPath = r.getBroadcaster().getID();
56              // Get rid of the AtmosphereFramework mapped path.
57              chatroomName = mappedPath.split("/")[2];
58              factory = r.getAtmosphereConfig().getBroadcasterFactory();
59          }
60  
61          return new ChatProtocol(users.keySet(), factory.lookupAll());
62      }
63  
64      /**
65       * Invoked when the client disconnect or when an unexpected closing of the underlying connection happens.
66       *
67       * @param event
68       */
69      @Disconnect
70      public void onDisconnect(AtmosphereResourceEvent event) {
71          if (event.isCancelled()) {
72              // We didn't get notified, so we remove the user.
73              users.values().remove(event.getResource().uuid());
74              logger.info("Browser {} unexpectedly disconnected", event.getResource().uuid());
75          } else if (event.isClosedByClient()) {
76              logger.info("Browser {} closed the connection", event.getResource().uuid());
77          }
78      }
79  
80      /**
81       * Simple annotated class that demonstrate how {@link org.atmosphere.config.managed.Encoder} and {@link org.atmosphere.config.managed.Decoder
82       * can be used.
83       *
84       * @param message an instance of {@link ChatProtocol }
85       * @return
86       * @throws IOException
87       */
88      @Message(encoders = {JacksonEncoder.class}, decoders = {ProtocolDecoder.class})
89      public ChatProtocol onMessage(ChatProtocol message) throws IOException {
90  
91          if (!users.containsKey(message.getAuthor())) {
92              users.put(message.getAuthor(), message.getUuid());
93              return new ChatProtocol(message.getAuthor(), " entered room " + chatroomName, users.keySet(), factory.lookupAll());
94          }
95  
96          if (message.getMessage().contains("disconnecting")) {
97              users.remove(message.getAuthor());
98              return new ChatProtocol(message.getAuthor(), " disconnected from room " + chatroomName, users.keySet(), factory.lookupAll());
99          }
100 
101         message.setUsers(users.keySet());
102         logger.info("{} just send {}", message.getAuthor(), message.getMessage());
103         return new ChatProtocol(message.getAuthor(), message.getMessage(), users.keySet(), factory.lookupAll());
104     }
105 
106     @Message(decoders = {UserDecoder.class})
107     public void onPrivateMessage(UserMessage user) throws IOException {
108         String userUUID = users.get(user.getUser());
109         if (userUUID != null) {
110             AtmosphereResource r = AtmosphereResourceFactory.getDefault().find(userUUID);
111 
112             if (r != null) {
113                 ChatProtocol m = new ChatProtocol(user.getUser(), " sent you a private message: " + user.getMessage().split(":")[1], users.keySet(), factory.lookupAll());
114                 if (!user.getUser().equalsIgnoreCase("all")) {
115                     factory.lookup(mappedPath).broadcast(m, r);
116                 }
117             }
118         } else {
119             ChatProtocol m = new ChatProtocol(user.getUser(), " sent a message to all chatroom: " + user.getMessage().split(":")[1], users.keySet(), factory.lookupAll());
120             MetaBroadcaster.getDefault().broadcastTo("/*", m);
121         }
122     }
